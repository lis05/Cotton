44 static char getCharacter(std::string::const_iterator &it, const std::string &str, ErrorManager *error_manager) {
45  char res;
46  if (*it == '\\') {
47  if (it + 1 == str.end()) {
48  error_manager->signalError("Expected a character", it + 1 - str.begin());
49  }
50  switch (it[1]) {
51  case 'a' : res = '\a'; break;
52  case 'b' : res = '\b'; break;
53  case 'f' : res = '\f'; break;
54  case 'n' : res = '\n'; break;
55  case 'r' : res = '\r'; break;
56  case 't' : res = '\t'; break;
57  case 'v' : res = '\v'; break;
58  case '\\' : res = '\\'; break;
59  case '\"' : res = '\"'; break;
60  case '\'' : res = '\''; break;
61  default : error_manager->signalError("Invalid character", it + 1 - str.begin());
62  }
63  it += 2;
64  }
65  else {
66  res = *it;
67  it++;
68  }
69 
70  return res;
71 }

73 static std::string parseString(const std::string &str, ErrorManager *error_manager) {
74  eassert(!str.empty(), "Empty string", error_manager);
75  std::string res;
76  // string
77  if (str.front() == '\"') {
78  auto it = str.begin() + 1;
79  while (it != str.end() && *it != '\"') {
80  res += getCharacter(it, str, error_manager);
81  }
82  eassert(*it == '\"', "Expected a string end", error_manager);
83  }
84  else {
85  auto it = str.begin() + 1;
86  eassert(it != str.end() && *it != '\'', "Expected a character", error_manager);
87  res += getCharacter(it, str, error_manager);
88  eassert(*it == '\'', "Expected a character end", error_manager);
89  }
90  return res;
91 }

353 static void skipString(Lexer &lexer, std::string::iterator &it, std::string &str) {
354  auto string_begin = it;
355  bool found_end = false;
356  it++;
357  while (it != str.end()) {
358  if (*it == '\"') {
359  // it may be \", so we have to check for that
360  int cnt = 0;
361  auto it2 = it - 1;
362  while (it2 != string_begin && *it2 == '\\') {
363  cnt++;
364  it2--;
365  }
366 
367  if (cnt % 2 == 0) {
368  // string end
369  found_end = true;
370  break;
371  }
372  }
373 
374  it++;
375  }
376  if (!found_end) {
377  lexer.error_manager->signalError("Expected the end of the string", it - str.begin());
378  }
379  it++;
380 }

382 static void skipCharacter(Lexer &lexer, std::string::iterator &it, std::string &str) {
383  it++;
384 
385  if (it == str.end()) {
386  lexer.error_manager->signalError("Expected a character", it - str.begin());
387  }
388 
389  if (*it == '\'') {
390  lexer.error_manager->signalError("Character cannot be empty", it - str.begin());
391  }
392 
393  if (*it == '\\') {
394  it++;
395  }
396 
397  if (it == str.end()) {
398  lexer.error_manager->signalError("Expected the end of the character", it - str.begin());
399  }
400 
401  it++;
402 
403  if (*it != '\'') {
404  lexer.error_manager->signalError("Expected the end of the character", it - str.begin());
405  }
406 
407  it++;
408 }

410 void Lexer::faze1FillComments(std::string &input) {
411  auto it = input.begin();
412  while (it != input.end()) {
413  if (*it == '\"') {
414  skipString(*this, it, input);
415  }
416  else if (*it == '\'') {
417  skipCharacter(*this, it, input);
418  }
419  // now we're safe, and can remove comments
420  else if (*it == '/' && it + 1 != input.end() && it[1] == '*') {
421  // everything until */ is a comment
422  while (it != input.end()) {
423  if (*it == '*' && it + 1 != input.end() && it[1] == '/') {
424  *it = ' ';
425  it[1] = ' ';
426  it += 2;
427  break;
428  }
429  *it = ' ';
430  it++;
431  }
432  }
433  else if (*it == '/' && it + 1 != input.end() && it[1] == '/') {
434  // everything until new line is a comment
435  while (it != input.end()) {
436  if (*it == '\n') {
437  *it = ' ';
438  it++;
439  break;
440  }
441  *it = ' ';
442  it++;
443  }
444  }
445 
446  else {
447  it++;
448  }
449  }
450 }

452 std::vector<Token> Lexer::faze2FormConnectedTokens(const std::string &input) {
453  std::string buf;
454  std::vector<Token> tokens;
455  auto it = input.begin();
456 
457  while (it != input.end()) {
458  if (*it == '"') {
459  if (!buf.empty()) {
460  tokens.push_back(Token(buf, it - input.begin() - buf.size(), it - input.begin(), ""));
461  buf.clear();
462  }
463  auto string_begin = it;
464  it++;
465  // since the string has passed the first faze, it has it's ending
466  while (*it != '"') {
467  getCharacter(it, input, this->error_manager);
468  }
469  this->error_manager->setErrorCharPos(it - input.begin());
470  eassert(*it == '\"', "String doesn't end with a double quote", this->error_manager);
471  it++;
472  auto string_end = it;
473 
474  it = string_begin;
475  while (it != string_end) {
476  buf += *it;
477  it++;
478  }
479 
480  tokens.push_back(Token(buf, it - input.begin() - buf.size(), it - input.begin(), ""));
481  tokens.back().id = Token::STRING_LIT;
482  buf.clear();
483  }
484  else if (*it == '\'') {
485  if (!buf.empty()) {
486  tokens.push_back(Token(buf, it - input.begin() - buf.size(), it - input.begin(), ""));
487  buf.clear();
488  }
489  auto char_begin = it;
490  it++;
491  // since the character has passed the first faze, it has it's ending
492  getCharacter(it, input, this->error_manager);
493  this->error_manager->setErrorCharPos(it - input.begin());
494  eassert(*it == '\'', "Character doesn't end with a single quote", this->error_manager);
495  it++;
496  auto char_end = it;
497 
498  it = char_begin;
499  while (it != char_end) {
500  buf += *it;
501  it++;
502  }
503 
504  tokens.push_back(Token(buf, it - input.begin() - buf.size(), it - input.begin(), ""));
505  tokens.back().id = Token::CHAR_LIT;
506  buf.clear();
507  }
508  else if (isspace(*it)) {
509  if (!buf.empty()) {
510  tokens.push_back(Token(buf, it - input.begin() - buf.size(), it - input.begin(), ""));
511  buf.clear();
512  }
513  it++;
514  }
515  else {
516  buf += *it;
517  it++;
518  }
519  }
520  if (!buf.empty()) {
521  tokens.push_back(Token(buf, it - input.begin() - buf.size(), it - input.begin(), ""));
522  buf.clear();
523  }
524  return tokens;
525 }

532 static void splitByOperator(const Token &token,
533  std::string::const_iterator &split_end,
534  std::string::const_iterator &it,

556 std::vector<Token> Lexer::faze3SplitByOperatorsAndSeparators(const std::vector<Token> &tokens) {
557  std::vector<Token> res;
558  for (auto &token : tokens) {
559  auto it = token.data.begin();
560  auto split_end = it;
561  while (it != token.data.end()) {
562  bool f = false;
563  for (auto &splitter : sortedSplitters) {
564  if (token.id != Token::STRING_LIT && token.id != Token::CHAR_LIT
565  && token.data.substr(it - token.data.begin(), splitter.size()) == splitter)
566  {
567  f = true;
568  splitByOperator(token, split_end, it, res, splitter);
569  break;
570  }
571  }
572  if (!f) {
573  it++;
574  }
575  }
576 
577  if (it != split_end) {
578  Token t(token.data.substr(split_end - token.data.begin(), it - split_end),
579  split_end - token.data.begin() + token.begin_pos,
580  it - token.data.begin() + token.begin_pos,
581  "");
582  res.push_back(t);
583  split_end = it;
584  }
585  }
586  return res;
587 }

589 std::vector<Token> Lexer::faze4FindRealNumbers(const std::vector<Token> &tokens) {
590  std::vector<Token> res;
591  for (auto &token : tokens) {
592  if (!token.data.empty() && token.data[0] != '\"' && token.data[0] != '\'') {
593  if (token.data.find(".") != token.data.npos) {
594  try {
595  size_t ptr;
596  std::stod(token.data, &ptr);
597  if (ptr == token.data.size()) {
598  res.push_back(token);
599  continue;
600  }
601  } catch (...) {
602  }
603 
604  // not real, splitting by dot operator
605  auto it = token.data.begin();
606  auto split_end = it;
607  bool f = false;
608  std::string splitter = ".";
609  while (it != token.data.end()) {
610  if (token.data.substr(it - token.data.begin(), splitter.size()) == splitter) {
611  f = true;
612  splitByOperator(token, split_end, it, res, splitter);
613  }
614  else {
615  it++;
616  }
617  }
618 
619  if (it != split_end) {
620  Token t(token.data.substr(split_end - token.data.begin(), it - split_end),
621  split_end - token.data.begin() + token.begin_pos,
622  it - token.data.begin() + token.begin_pos,
623  "");
624  res.push_back(t);
625  split_end = it;
626  }
627  }
628  else {
629  res.push_back(token);
630  }
631  }
632  else {
633  res.push_back(token);
634  }
635  }
636  return res;
637 }

650  this->faze1FillComments(input);
651  std::vector<Token> tokens = this->faze2FormConnectedTokens(input);
