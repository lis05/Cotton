126  rt->verifyHasMethod(arg, MagicMethods::mm__repr__(rt), rt->getContext().sub_areas[index]);
127

129  rt->newContext();
130  rt->getContext().area = ta;
131  rt->getContext().sub_areas = {ta, ta};
132  auto res = rt->runMethod(MagicMethods::mm__repr__(rt), arg, {arg}, true);
133  CF_printraw({res}, rt, false);
134  rt->popContext();
135  }
136  return rt->protected_nothing;
