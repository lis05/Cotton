65 Object *Type::getMethod(int64_t id, Runtime *rt) {
66  ProfilerCAPTURE();
67  auto it = this->methods.find(id);
68  if (it != this->methods.end()) {
69  rt->verifyIsValidObject(it->second);
70  return it->second;
71  }
72  rt->signalError(this->userRepr(rt) + " doesn't have method " + rt->nds->userRepr(id), rt->getContext().area);
