232  ProfilerCAPTURE();

234  rt->verifyIsValidObject(arg, rt->getContext().sub_areas[1]);
235 
236  if (!rt->isOfType(arg, rt->character_type)) {
237  return rt->protected_false;
238  }
239 
240  if (rt->isInstanceObject(self, rt->character_type)) {
241  if (!rt->isInstanceObject(arg, rt->character_type)) {
242  return rt->protected_false;
243  }
244  return rt->protectedBoolean(getCharacterValueFast(self) == getCharacterValueFast(arg));
245  }
246  else if (rt->isTypeObject(self, rt->character_type)) {
247  if (!rt->isTypeObject(arg, rt->character_type)) {
248  return rt->protected_false;
249  }
250  return rt->protected_true;
251  }
252 
253  return rt->protected_false;
