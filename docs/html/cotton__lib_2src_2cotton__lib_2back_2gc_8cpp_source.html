79 void GCDefaultStrategy::acknowledgeEndOfCycle(Runtime *rt) {
80  ProfilerCAPTURE();
81  this->prev_num_tracked
82  = rt->gc->tracked_instances.size() + rt->gc->tracked_objects.size() + rt->gc->tracked_types.size();
83  this->prev_sizeof_tracked = 0;
84  for (auto &[obj, _] : rt->gc->tracked_objects) {
85  this->prev_sizeof_tracked += sizeof(obj);
86  }
87  for (auto &[ins, _] : rt->gc->tracked_instances) {
88  this->prev_sizeof_tracked += ins->getSize();
89  }
90  for (auto &[type, _] : rt->gc->tracked_types) {
91  this->prev_sizeof_tracked += sizeof(type);
92  }
93  this->num_tracked = this->prev_num_tracked;
94  this->sizeof_tracked = this->prev_sizeof_tracked;
95 }

102 void GCDefaultStrategy::checkConditions(Runtime *rt) {
103  ProfilerCAPTURE();
104  if (this->sizeof_tracked >= MIN_CYCLE_SIZE
105  && ((this->prev_num_tracked < this->num_tracked / NUM_TRACKED_MULT)
106  || (this->prev_sizeof_tracked < this->sizeof_tracked / SIZEOF_TRACKED_MULT)
107  || (this->ops_cnt >= OPS_MOD)))
108  {
109  this->ops_cnt %= OPS_MOD;
110  rt->gc->runCycle(rt);
111  }
112 }

257 void GC::runCycle(Runtime *rt) {
258  ProfilerCAPTURE();
259  if (!this->enabled) {
260  return;
261  }
262  // mark
263  auto scope = rt->scope;
264  while (scope != NULL) {
265  for (auto &[_, obj] : scope->variables) {
266  mark(obj, rt);
267  }
268  for (auto &obj : scope->arguments) {
269  mark(obj, rt);
270  }
271  scope = scope->prev;
272  }
273  for (auto &[obj, _] : this->held_objects) {
274  mark(obj, rt);
275  }
276  for (auto &[_, obj] : rt->globals) {
277  mark(obj, rt);
278  }
279  // sweep
280  std::vector<Object *> deleted_objects;
281  for (auto &[obj, _] : this->tracked_objects) {
282  if (obj->gc_mark != this->gc_mark) {
283  deleted_objects.push_back(obj);
284  }
285  }
286  for (auto &obj : deleted_objects) {
287  this->tracked_objects.erase(obj);
288  delete (obj);
289  }
290  std::vector<Instance *> deleted_instances;
291  for (auto &[ins, _] : this->tracked_instances) {
292  if (ins->gc_mark != this->gc_mark) {
293  deleted_instances.push_back(ins);
294  }
295  }
296  for (auto &ins : deleted_instances) {
297  this->tracked_instances.erase(ins);
298  delete (ins);
299  }
300  std::vector<Type *> deleted_types;
301  for (auto &[type, _] : this->tracked_types) {
302  if (type->gc_mark != this->gc_mark) {
303  deleted_types.push_back(type);
304  }
305  }
306  for (auto &type : deleted_types) {
307  this->tracked_types.erase(type);
308  delete type;
309  }
310 
311  this->gc_mark = !this->gc_mark;
312  this->gc_strategy->acknowledgeEndOfCycle(rt);
313  // fprintf(stderr, "GC CYCLE END\n");
314 }
