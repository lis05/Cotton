93  rt->verifyIsInstanceObject(self, rt->array_type, rt->getContext().sub_areas[0]);

96  rt->verifyIsInstanceObject(arg, rt->integer_type, rt->getContext().sub_areas[1]);
97 
98  if (!(0 <= getIntegerValueFast(arg) && getIntegerValueFast(arg) < getArrayDataFast(self).size())) {
99  rt->signalError("Index " + arg->userRepr(rt) + " is out of array " + self->userRepr(rt) + " range",
100  rt->getContext().sub_areas[1]);
101  }
102  return getArrayDataFast(self)[getIntegerValueFast(arg)];

105 static Object *ArrayEqAdapter(Object *self, Object *arg, Runtime *rt, bool execution_result_matters) {
106  ProfilerCAPTURE();
107  rt->verifyIsOfType(self, rt->array_type, rt->getContext().sub_areas[0]);
108  rt->verifyIsValidObject(arg, rt->getContext().sub_areas[1]);
109 
110  if (!execution_result_matters) {
111  return rt->protected_false;
112  }
113 
114  if (!rt->isOfType(arg, rt->string_type)) {
115  return rt->protected_false;
116  }
117 
118  if (rt->isInstanceObject(self, rt->integer_type)) {
119  if (!rt->isInstanceObject(arg, rt->integer_type)) {
120  return rt->protected_false;
121  }
122  auto a1 = getArrayDataFast(self);
123  auto a2 = getArrayDataFast(self);
124  if (a1.size() != a2.size()) {
125  return rt->protected_false;
126  }
127  for (int64_t i = 0; i < a1.size(); i++) {
128  auto &ta = rt->getContext().area;
129  rt->newContext();
130  rt->getContext().area = ta;
131  rt->getContext().sub_areas = {ta, ta};
132  auto res = rt->runOperator(OperatorNode::EQUAL, a1[i], a2[i], true);
133  rt->popContext();
134  if (!getBooleanValueFast(res)) {
135  return rt->protected_false;
136  }
137  }
138  return rt->protected_true;
139  }
140  else if (rt->isTypeObject(self, rt->integer_type)) {
141  if (!rt->isTypeObject(arg, rt->integer_type)) {
142  return rt->protected_false;
143  }
144  return rt->protected_true;
145  }
146 
147  return rt->protected_false;
148 }

169 static Object *arrayResizeMethod(const std::vector<Object *> &args, Runtime *rt, bool execution_result_matters) {
170  ProfilerCAPTURE();
171  rt->verifyExactArgsAmountMethod(args, 1);
172  auto self = args[0];
173  auto new_size = args[1];
174  rt->verifyIsInstanceObject(self, rt->array_type, rt->getContext().sub_areas[0]);
175  rt->verifyIsInstanceObject(new_size, rt->integer_type, rt->getContext().sub_areas[1]);
176 
177  int64_t oldn = getArrayDataFast(self).size();
178  int64_t newn = getIntegerValueFast(new_size);
179  if (newn <= 0) {
180  rt->signalError("New array size must be positive: " + new_size->userRepr(rt),
181  rt->getContext().sub_areas[1]);
182  }
183  getArrayDataFast(self).resize(newn);
184 
185  for (int64_t i = oldn; i < newn; i++) {
186  getArrayDataFast(self)[i] = makeNothingInstanceObject(rt);
187  }
188 
189  return self;
190 }

192 static Object *mm__repr__(const std::vector<Object *> &args, Runtime *rt, bool execution_result_matters) {
193  ProfilerCAPTURE();
194  rt->verifyExactArgsAmountMethod(args, 0);
195  auto self = args[0];
196  rt->verifyIsOfType(self, rt->array_type, rt->getContext().sub_areas[1]);
197 
198  if (!execution_result_matters) {
199  return self;
200  }
201 
202  if (rt->isTypeObject(self, NULL)) {
203  return makeStringInstanceObject("Array", rt);
204  }
205 
206  auto &arr = getArrayDataFast(self);
207  std::string res = "{";
208  if (arr.size() != 0) {
209  auto o = rt->runMethod(MagicMethods::mm__repr__(rt), arr[0], {arr[0]}, true);
210  rt->verifyIsInstanceObject(o, rt->string_type, rt->getContext().sub_areas[1]);
211  res += getStringDataFast(o);
212  }
213 
214  for (int64_t i = 1; i < arr.size(); i++) {
215  auto o = rt->runMethod(MagicMethods::mm__repr__(rt), arr[i], {arr[i]}, true);
216  rt->verifyIsInstanceObject(o, rt->string_type, rt->getContext().sub_areas[1]);
217  res += ", " + getStringDataFast(o);
218  }
219 
220  res += "}";
221 
222  return makeStringInstanceObject(res, rt);
