740  passert(token != NULL, "Failed to get operator from a NULL token");

742  case Token::PLUS_PLUS_OP :
743  return (is_pre_op) ? OperatorInfo {OperatorNode::PRE_PLUS_PLUS, 2, RIGHT_TO_LEFT}
744  : OperatorInfo {OperatorNode::POST_PLUS_PLUS, 1, LEFT_TO_RIGHT};

840 bool Parser::canPassThrough(OperatorInfo op_info) {
841  if (this->state.cur_priority > op_info.priority) {
842  // our operator is of higher priority, therefore we can go through
843  return true;
844  }
845  if (this->state.cur_priority == op_info.priority) {
846  // our operator is of the same priority, so associativity determines the result
847  //
848  // ex: associativity = LEFT_TO_RIGHT: 1 + 2 + 3 is (1 + 2) + 3, so no passing through
849  // ^ ^
850  // but associativity = RIGHT_TO_LEFT: a = b = c is a = (b = c), so we pass through
851  // ^ ^
852  return op_info.associativity == RIGHT_TO_LEFT;
853  }
854  // otherwise, our operator is of lower priority, so we may not go through it
855  return false;
856 }

966  return ParsingResult("", this);

998  return ParsingResult("Expected an identifier", this);
999  }

1001  this->consume();
1002
