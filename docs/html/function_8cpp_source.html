73 static Object *
74 FunctionCallAdapter(Object *self, const std::vector<Object *> &args, Runtime *rt, bool execution_result_matters) {
75  ProfilerCAPTURE();
76  rt->verifyIsInstanceObject(self, rt->function_type, rt->getContext().sub_areas[0]);
77  auto f = icast(self->instance, FunctionInstance);
78  if (f->is_internal) {
79  if (f->internal_ptr == NULL) {
80  rt->signalError("Failed to execute NULL internal function: " + self->userRepr(rt),
81  rt->getContext().area);
82  }
83  auto res = f->internal_ptr(args, rt, execution_result_matters);
84  if (execution_result_matters && res == NULL) {
85  rt->signalError("Execution of internal function " + self->userRepr(rt) + " has failed",
86  rt->getContext().area);
87  }
88  return res;
89  }
90  else {
91  if (f->cotton_ptr == NULL || f->cotton_ptr->body == NULL) {
92  rt->signalError("Failed to execute NULL function " + self->userRepr(rt), rt->getContext().area);
93  }
94  rt->newFrame(false);
95  // rt->scope->arguments.push_back(self); // is it needed?
96  for (auto arg : args) {
97  rt->scope->arguments.push_back(arg);
98  }
99  if (f->cotton_ptr->params != NULL) {
100  int i = 0;
101  for (auto token : f->cotton_ptr->params->list) {
102  if (i >= args.size()) {
103  rt->scope->addVariable(token->nameid, makeNothingInstanceObject(rt), rt);
104  continue;
105  }
106  rt->scope->addVariable(token->nameid, args[i], rt);
107  i++;
108  }
109  }
110  auto res = rt->execute(f->cotton_ptr->body, execution_result_matters);
111  rt->popFrame();
112  if (execution_result_matters && res == NULL) {
113  rt->signalError("Execution of function " + self->userRepr(rt) + " has failed",
114  rt->getContext().sub_areas[0]);
115  }
116  return res;
117  }
118 }

120 static Object *FunctionEqAdapter(Object *self, Object *arg, Runtime *rt, bool execution_result_matters) {
121  ProfilerCAPTURE();
122  rt->verifyIsOfType(self, rt->function_type, rt->getContext().sub_areas[0]);
123  rt->verifyIsValidObject(arg, rt->getContext().sub_areas[1]);
124 
125  if (!rt->isOfType(arg, rt->function_type)) {
126  return rt->protected_false;
127  }
128 
129  if (rt->isInstanceObject(self, rt->function_type)) {
130  if (!rt->isInstanceObject(arg, rt->function_type)) {
131  return rt->protected_false;
132  }
133  auto f1 = icast(self, FunctionInstance);
134  auto f2 = icast(arg, FunctionInstance);
135  if (f1->is_internal && f2->is_internal) {
136  return rt->protectedBoolean(f1->internal_ptr == f2->internal_ptr);
137  }
138  else if (!f1->is_internal && !f2->is_internal) {
139  return rt->protectedBoolean(f1->cotton_ptr == f2->cotton_ptr);
140  }
141  return rt->protected_false;
142  }
143  else if (rt->isTypeObject(self, rt->function_type)) {
144  if (!rt->isTypeObject(arg, rt->function_type)) {
145  return rt->protected_false;
146  }
147  return rt->protected_true;
148  }
149 
150  return rt->protected_false;
151 }
